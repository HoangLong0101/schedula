2.2  Why These Packages
Package	Rationale
flutter_bloc	Official BLoC library. Clean separation of events/states, excellent testability via bloc_test. Easier to onboard new devs compared to Riverpod.
injectable + get_it	injectable generates DI wiring from annotations � no manual GetIt.registerFactory() for every class. Scales cleanly as codebase grows.
go_router	Official Flutter navigation package. Supports deep links, typed routes, and auth guards. Preferred over AutoRoute for new projects.
dartz (Either)	UseCases return Either<Failure, T>. BLoC maps Left to error state, Right to success state. Replaces try/catch exception flows in the domain layer.
flutter_secure_storage	Stores FCM tokens and auth tokens in iOS Keychain / Android Keystore. Never use SharedPreferences for tokens.
table_calendar	Best-maintained Flutter calendar widget. Supports multi-event day markers � exactly what the booking grid needs.

       .where('tenantId', isEqualTo: tenantId)
       .orderBy('startTime')
       .snapshots()
       .map((s) => s.docs
         .map((d) => BookingModel.fromFirestore(d).toEntity())
         .toList());

3.3  Firebase Auth � Roles as Custom Claims
Store roles (owner / receptionist / staff) as custom claims on the Firebase Auth JWT � not in Firestore. Security Rules can then check roles without an extra database read on every request.

Cloud Function � assign role (deploy to all 3 projects)
// functions/src/auth/setUserRole.ts
import * as admin from 'firebase-admin';
import * as functions from 'firebase-functions';
 
export const setUserRole = functions.https.onCall(async (data, context) => {
  if (context.auth?.token?.role !== 'owner') {
    throw new functions.https.HttpsError('permission-denied', 'Owners only');
  }
  const { uid, role, tenantId } = data;
  if (!['owner','receptionist','staff'].includes(role)) {
    throw new functions.https.HttpsError('invalid-argument', 'Bad role');
  }
  await admin.auth().setCustomUserClaims(uid, { role, tenantId });
  return { success: true };
});

Flutter � read claims after sign-in
// Force-refresh the token so custom claims are visible immediately
final token = await FirebaseAuth.instance.currentUser!

3.4  Firestore Security Rules
Deploy these rules before writing any feature code. The rules enforce role access and tenant isolation simultaneously.

// firestore.rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
 
    function isAuth()           { return request.auth != null; }
    function role()             { return request.auth.token.get('role','staff'); }
    function myTenant()         { return request.auth.token.get('tenantId',''); }
    function sameTenant(docTid) { return myTenant() == docTid; }
    function isOwner()          { return role() == 'owner'; }
    function canBook()          { return role() == 'owner' || role() == 'receptionist'; }
 
    match /bookings/{id} {
      allow read: if isAuth() && sameTenant(resource.data.tenantId);
      allow create: if isAuth() && canBook()
                    && sameTenant(request.resource.data.tenantId)
                    && request.resource.data.keys().hasAll([
                         'tenantId','staffId','customerId',
                         'serviceId','startTime','endTime','status']);
      allow update: if isAuth() && sameTenant(resource.data.tenantId)
                    && (canBook() ||
                       (role() == 'staff'
                        && request.auth.uid == resource.data.staffId
                        && request.resource.data.diff(resource.data)
                             .affectedKeys().hasOnly(['status','updatedAt'])));
      allow delete: if isOwner() && sameTenant(resource.data.tenantId);
    }
    match /customers/{id} {
      allow read:  if isAuth() && sameTenant(resource.data.tenantId);
      allow write: if isAuth() && canBook()
                   && sameTenant(request.resource.data.tenantId);
    }
    match /users/{userId} {
      allow read:  if isAuth()
                   && (request.auth.uid == userId
                       || (isOwner() && sameTenant(resource.data.tenantId)));
      allow write: if isOwner() && sameTenant(request.resource.data.tenantId);
    }
    match /services/{id} {
      allow read:  if isAuth() && sameTenant(resource.data.tenantId);
      allow write: if isOwner() && sameTenant(request.resource.data.tenantId);
    }
    match /slots/{id} {
      allow read:  if isAuth() && sameTenant(resource.data.tenantId);
      allow write: if isAuth() && canBook()
                   && sameTenant(request.resource.data.tenantId);
    }
    match /notifications/{id} {
      allow read:  if isAuth() && sameTenant(resource.data.tenantId);
      allow write: if false; // Cloud Functions only � never from client
    }
  }
}

// lib/core/services/notification_service.dart
import 'package:cloud_firestore/cloud_firestore.dart';
import 'package:firebase_auth/firebase_auth.dart';
import 'package:firebase_messaging/firebase_messaging.dart';
 
// Must be a top-level function (not inside a class)
@pragma('vm:entry-point')
Future<void> _bgHandler(RemoteMessage message) async {
  // Keep lightweight � no UI here
}
 
class NotificationService {
  final _fcm = FirebaseMessaging.instance;
 
  Future<void> initialize() async {
    FirebaseMessaging.onBackgroundMessage(_bgHandler);
 
    await _fcm.requestPermission(alert: true, badge: true, sound: true);
 
    final token = await _fcm.getToken();
    if (token != null) await _saveToken(token);
    _fcm.onTokenRefresh.listen(_saveToken);
 
    FirebaseMessaging.onMessage.listen((msg) {
      // TODO: show in-app snackbar or banner
    });
  }
 
  Future<void> _saveToken(String token) async {
    final uid = FirebaseAuth.instance.currentUser?.uid;
    if (uid == null) return;
    await FirebaseFirestore.instance.collection('users').doc(uid).update({
      'fcmToken': token,
      'fcmUpdatedAt': FieldValue.serverTimestamp(),
    });
  }
}
